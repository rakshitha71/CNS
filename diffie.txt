#include<stdio.h>
int power(int a,int b,int n){
    if(b==0)
        return 1;
    long long int temp = power(a,b/2,n)%n;
    long long int result=(temp*temp)%n;
   if(b%2==1)
    result=(result*a)%n;
  return result;
}

{
